----------------------
MathLibrary.cpp
----------------------
#include "MathLibrary.h"

int add(int a, int b) {
    return a + b;
}

int subtract(int a, int b) {
    return a - b;
}

int multiply(int a, int b) {
    return a * b;
}

int divide(int a, int b) {
    return a / b;
}

int square(int a) {
    return a * a;
}

float percent(float a, float b) {
    return (a / b) * 100;
}

----------------------
MathLibrary.h
----------------------
#pragma once

#ifdef __cplusplus
extern "C" {
#endif

int add(int a, int b);
int subtract(int a, int b);
int multiply(int a, int b);
int divide(int a, int b);
int square(int a);
float percent(float a, float b);

#ifdef __cplusplus
}
#endif
